
ENTITY: Department
Feature: Create department
API: POST /api/departments
DB Action: INSERT into departments

Feature: View all departments
API: GET /api/departments
DB Action: SELECT from departments


ENTITY: Employee
Feature: Add employee
API: POST /api/employees
DB Action: INSERT into employees

Feature: View all employees
API: GET /api/employees
DB Action: SELECT from employees

Feature: View single employee
API: GET /api/employees/{employee_id}
DB Action: SELECT by employee_id

Feature: Update employee
API: PUT /api/employees/{employee_id}
DB Action: UPDATE employees

Feature: Delete employee
API: DELETE /api/employees/{employee_id}
DB Action: DELETE from employees


ENTITY: Asset
Feature: Add asset
API: POST /api/assets
DB Action: INSERT into assets

Feature: View all assets
API: GET /api/assets
DB Action: SELECT from assets

Feature: Update asset status
API: PUT /api/assets/{asset_id}
DB Action: UPDATE assets


ENTITY: Asset Assignment
Feature: Assign asset to employee
API: POST /api/asset-assignments
DB Action:
- INSERT into asset_assignments
- UPDATE assets.status = 'Assigned'

Feature: View all assignments
API: GET /api/asset-assignments
DB Action: SELECT from asset_assignments


ENTITY: Asset Maintenance
Feature: Log maintenance
API: POST /api/asset-maintenance
DB Action: INSERT into asset_maintenance_logs

Feature: View maintenance history
API: GET /api/asset-maintenance/{asset_id}
DB Action: SELECT by asset_id
